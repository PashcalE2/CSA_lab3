org 0
v0_input: word write_handler
v1_output: word read_handler

org 0x20
msg1: byte 'Whats your name? ',0
msg2: byte 'Hello, ',0
msg3: byte '!',0

current: byte 0
output_str: byte 256 dup(0)


org 0x1000
start:
    ei

    push output_str      ; sum of strings
    push msg1            ; additional string
    call add_to_str      ; push ip, add_to_str => ip
    add byte [current], r1  ; we added r1 chars
    pop r1
    pop r1


    mov r1, output_str      ; calc sum-string last char pointer
    add byte r1, [current]

    push r1              ; sum of strings
    push msg2            ; additional string
    call add_to_str      ; push ip, add_to_str => ip
    add byte [current], r1  ; we added r1 chars
    pop r1
    pop r1


    mov r1, output_str      ; calc sum-string last char pointer
    add byte r1, [current]

    push r1              ; sum of strings
    call read
    dec r1
    add byte [current], r1  ; we read r1 chars
    pop r1


    mov r1, output_str
    add byte r1, [current]

    push r1
    push msg3
    call add_to_str
    add byte [current], r1 ; we added r1 chars
    pop r1
    pop r1

    push output_str
    call write
    pop r1

    halt


add_to_str:
    mov r7, [sp+4] ; add_base
    mov r6, [sp+2] ; additional_base
    mov r5, 0 ; counter

.loop:
    cmp byte [r6+r5], 0
    jz add_to_str.ret
    mov [r7+r5], [r6+r5]
    inc r5
    jmp add_to_str.loop

.ret:
    mov r1, r5
    ret


write:
    mov r7, [sp+2] ; base
    mov r6, 0
.wait:
    mov byte r1, [0x12]
    cmp byte r1, 0
    jne write.int
    jmp write.wait
.int:
    int 0
    cmp byte [r7+r6], 0
    jz write.ret
    inc r6
    jne write.wait

.ret:
    ret

write_handler:
    mov byte [0x13], [r6+r7]
    mov byte [0x12], 0
    iret


read:
    mov r7, [sp+2] ; base
    mov r6, 0
.wait:
    cmp byte [0x10], 0 ; check flag
    jz read.wait ;

    int 1
    inc r6
    cmp byte [r7+r6*1-1], 0
    jz read.ret
    jmp read.wait
.ret:
    mov r1, r6
    ret

read_handler:
    mov byte [r7+r6], [0x11] ; save to buffer
    mov byte [0x10], 0
    iret
